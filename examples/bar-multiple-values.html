// Use Morris.Bar
Morris.Bar({
  element: 'graph',
  data: [
    {x: '2011 Q1', y: {a: 7, b: 3}, z: {a: 3, b: 4} },
    {x: '2011 Q2', y: {a: 6, b: 2}, z: {a: 4, b: 2} },
    {x: '2011 Q3', y: {a: 5, b: 1}, z: {a: 5, b: 3} }
  ],
  xkey: 'x',
  ykeys: ['y', 'z'],
  zkeys: ['a', 'b'],
  labels: ['Y', 'Z'],
  zLabels: ['A', 'B']
});
