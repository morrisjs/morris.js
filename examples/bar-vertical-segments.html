Bar charts


// Use Morris.Bar
Morris.Bar({
      element: 'graph',
      data: [
        {x: 'foo', y: 2, z: 3},
        {x: 'bar', y: 4, z: 6},
        {x: 'fizz', y: 5, z: 1},
        {x: 'buzz', y: 0, z: 5},
        {x: 'ku', y: 2, z: 3},
        {x: 'qu', y: 4, z: 6},
        {x: 'lua', y: 5, z: 1},
        {x: 'mua', y: 0, z: 5}
      ],
      xkey: 'x',
      ykeys: ['y', 'z'],
      labels: ['Y', 'Z'],
      verticalGridCondition: function(index) {return index % 2},
      verticalGridColor: '#888888',
      verticalGridOpacity: '0.2'
});
