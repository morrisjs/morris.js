Pie Chart

var pie = Morris.Pie({
  element: 'graph',
  data: [
    {value: 70, label: 'foo'},
    {value: 5, label: 'bar'},
    {value: 10, label: 'baz'},
    {value: 5, label: 'A really really long label'}
  ],
  formatter: function (x) { return x + "%"},
  showLabel: false,
  drawOut: 15
});

pie.on("hover", function(segmentId, segmentData){
  console.log("hover", segmentId, segmentData);
});

pie.on("click", function(segmentId, segmentData){
  console.log("click", segmentId, segmentData);
});
